# Prep consists of:
# 1) filter to include only POSNEG trials
# 2) make chose pos var (0 = neg choice, 1 = pos choice)
# 3) z score RT within pts
# 4) bin affect and pos-neg self belief for plotting
# 5) renamed some vars so looks cleaner


#---PREP CONGRUENCY---#
df_cong <- cong %>%
    filter(delta_valence != 0) %>%
    mutate(
        chose_positive = as.integer(
            (delta_valence == -1 & response_0_1 == 0) |
                (delta_valence == 1 & response_0_1 == 1)
        ),
        affect_z = affect_z_emalevel,
        delta_sb = delta_self_belief,
        affect_bin = cut(affect_z,
            breaks = c(-Inf, -0.5, 0.5, Inf),
            labels = c("Unpleasant", "Neutral", "Pleasant")
        ),
        delta_posneg_sb = case_when(
            delta_valence == 1 ~ delta_sb, # NEGPOS trial (rightside posword leftside neg word)
            delta_valence == -1 ~ -delta_sb # POSNEG trial (rightside negword leftside pos word)
        ),
        delta_sb_bin = cut(delta_posneg_sb,
            breaks = seq(-0.5, 0.5, by = 0.1),
            labels = c(
                "-100 to -80", "-80 to -60", "-60 to -40", "-40 to -20", "-20 to 0",
                "0 to 20", "20 to 40", "40 to 60", "60 to 80", "80 to 100"
            ),
            include.lowest = TRUE
        )
    ) %>%
    group_by(subject_id) %>% # have to group df by subj b4 zscoring (then ungroup after)
    mutate(rt_z = as.numeric(scale(rt))) %>%
    ungroup()

#-------------------------------------------------------------------------------------------------
# H1a.MODEL1: MOOD CONGRUENCY: CHOICE MODEL
#-------------------------------------------------------------------------------------------------

# --- Hypothesized theoretical mechanism:
# When a participant feels more pleasant than their average, they will MORE efficiently retrieve positive (vs. negative) self-referential information from memory, THEREFORE they will be MORE likely to identify positive vs. negative words as self-descriptive (HIGHER P(CHOSE POS))
# When they feel more unpleasant than their average, they will LESS efficiently retrieve positive (vs. negative) self-referential information from memory, THEREFORE they will be LESS likely to identify positive vs. negative words as self-descriptive (LOWER P(CHOSE POS))

## H1a. Mood Congruency: RT Model 

#-------------------------------------------------------------------------------------------------
# H1a.MODEL2: MOOD CONGRUENCY: RT MODEL
#-------------------------------------------------------------------------------------------------

# When a pt feels more pleasant than average, they will sample more efficiently from pos vs. neg  self-schemas. If the pt also happens to be deciding on a trial where the pos word is more self-descriptive than the neg word (delta_posneg_sb > 0 ) this ends up being ANOTHER additive affect on beta_delta_valence which will lead to faster RTs on those trials. If the negative word is more self-descriptive than the pos word (delta_posneg_sb < 0 ) , this makes the decision even more difficult, and RTs shower.

# When a pt feels more unpleasant than average, they will sample less efficiently from pos vs. neg  self-schemas. If the pt also happens to be deciding on a trial where the pos word is more self-descriptive than the neg word (delta_posneg_sb > 0 ) this ends up further reducing beta_delta_valence (as if its) aka the deicsion is harder and RTs are slower.

# Key point to keep in mind!! RT depends on the absolute magnitude of drift rate. RT is slowest when beta_delta_valence is at its lowest ABSOLUTE value (aka when beta_delta_valence and delta_self_belief  cancel out). RT is fastest when beta_delta_valence is at its highest ABSOLUTE value (aka when beta_delta_valence and self-belief are additive).

        - In cases where participants are deciding between a positive and negative word where the NEGATIVE word is more self descriptive, as their affect becomes more UNPLEASANT than their average, they will ALSO take relatively LONGER to make the choice. 

        - In cases where participants are deciding between a positive and negative word where the POSITIVE word is more self descriptive, as their affect becomes more UNPLEASANT than their average, they will ALSO take relatively SHORTER to make the choice.
